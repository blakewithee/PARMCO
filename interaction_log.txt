/*
 * RPM monitoring thread - continuously calculates RPM
 * (This function is unchanged from your original)
 */
void* rpmThread(void* arg) {
    int last_state = -1;
    uint32_t pulse_times[1000];
    unsigned long pulse_count = 0;
    unsigned long pulse_index = 0;
    int iteration_count = 0;
    last_state = gpioRead(IR_SENSOR_PIN);
    while (!g_quit) {
        int current_state = gpioRead(IR_SENSOR_PIN);
        if (last_state != -1 && last_state != current_state) {
            uint32_t current_time = gpioTick();
            g_pulse_count++;
            pulse_times[pulse_index] = current_time;
            pulse_index = (pulse_index + 1) % 1000;
            if (pulse_count < 1000) {
                pulse_count++;
            }
        }
        last_state = current_state;
        iteration_count++;
        static uint32_t last_update = 0;
        uint32_t current_time = gpioTick();
        if (last_update == 0) {
            last_update = current_time;
        }
        uint32_t elapsed = current_time - last_update;
        if (current_time < last_update) {
            elapsed = (0xFFFFFFFF - last_update) + current_time;
        }
        if (elapsed >= (RPM_UPDATE_INTERVAL_MS * 1000)) {
            if (pulse_count > 0) {
                uint32_t one_second_ago = current_time - (RPM_CALCULATION_WINDOW_MS * 1000);
                unsigned long pulses_in_last_second = 0;
                unsigned long count_to_check = (pulse_count < 1000) ? pulse_count : 1000;
                for (unsigned long i = 0; i < count_to_check; i++) {
                    unsigned long idx = (pulse_index + 1000 - count_to_check + i) % 1000;
                    uint32_t pulse_time = pulse_times[idx];
                    if (current_time >= one_second_ago) {
                        if (pulse_time >= one_second_ago && pulse_time <= current_time) {
                            pulses_in_last_second++;
                        }
                    } else {
                        if (pulse_time >= one_second_ago || pulse_time <= current_time) {
                            pulses_in_last_second++;
                        }
                    }
                }
                if (pulses_in_last_second > 0) {
                    double edges_per_second = (double)pulses_in_last_second;
                    double blades_per_second = edges_per_second / 2.0;
                    double revolutions_per_second = blades_per_second / NUM_BLADES;
                    double rpm = revolutions_per_second * 60.0;
                    pthread_mutex_lock(&g_rpm_mutex);
                    g_current_rpm = rpm;
                    pthread_mutex_unlock(&g_rpm_mutex);
                } else {
                    pthread_mutex_lock(&g_rpm_mutex);
                    g_current_rpm = 0.0;
                    pthread_mutex_unlock(&g_rpm_mutex);
                }
            } else {
                pthread_mutex_lock(&g_rpm_mutex);
                g_current_rpm = 0.0;
                pthread_mutex_unlock(&g_rpm_mutex);
            }
            last_update = current_time;
        }
        if (++iteration_count >= 10000) {
            iteration_count = 0;
            usleep(0);
        }
    }
    return NULL;
}

// <-- NEW: Function to open the pipe
void openPipe() {
    // Open the pipe in non-blocking mode so it doesn't halt the program
    // if the python script isn't running yet.
    g_pipe_fd = open(FIFO_PATH, O_RDONLY | O_NONBLOCK);
    if (g_pipe_fd == -1) {
        if (errno == ENXIO) {
             printf("Waiting for BLE server to open pipe...\n");
        }
        return;
    }
    // We got a connection! Wrap it in a C stream (FILE*)
    g_pipe_stream = fdopen(g_pipe_fd, "r");
    if (g_pipe_stream == NULL) {
        close(g_pipe_fd);
        g_pipe_fd = -1;
        perror("fdopen");
        return;
    }
    printf("\n-> Named pipe is now open! Ready for BLE commands.\n");
}

void processCommand(char* input) {
    int new_speed;
    // Remove newline character
    input[strcspn(input, "\n")] = 0;
    input[strcspn(input, "\r")] = 0; // Also remove carriage return
    if (strlen(input) == 0) return; // Ignore empty commands
    printf("-> Command received: [%s]\n", input);
    // (Command parsing logic is identical to your original)
    if (strcmp(input, "on") == 0) motorOn();
    else if (strcmp(input, "off") == 0) motorOff();
    else if (strcmp(input, "+") == 0) {
        new_speed = g_speed + 10;
        if (new_speed > 100) new_speed = 100;
        setSpeed(new_speed);
    }
    else if (strcmp(input, "-") == 0) {
        new_speed = g_speed - 10;
        if (new_speed < 0) new_speed = 0;
        setSpeed(new_speed);
    }
    else if (input[0] == 's' && input[1] == ' ') {
        if (sscanf(input, "s %d", &new_speed) == 1) {
            if (new_speed >= 0 && new_speed <= 100) setSpeed(new_speed);
            else printf("ERROR: Speed must be 0-100\n");
        } else printf("ERROR: Use 's 50' for 50%%\n");
    }
    else if (strcmp(input, "f") == 0) setDirection(1);
    else if (strcmp(input, "r") == 0) setDirection(0);
    else if (strcmp(input, "rpm") == 0) {
        pthread_mutex_lock(&g_rpm_mutex);
        double rpm = g_current_rpm;
        pthread_mutex_unlock(&g_rpm_mutex);
        printf("-> Current RPM: %.2f\n", rpm);

    int max_fd;

        fd_set readfds;
        struct timeval timeout;
        FD_ZERO(&readfds);
        FD_SET(STDIN_FILENO, &readfds); // Always listen to keyboard
        max_fd = STDIN_FILENO;
        if (g_pipe_fd != -1) {
            // If pipe is open, listen to it
            FD_SET(g_pipe_fd, &readfds);
            if (g_pipe_fd > max_fd) max_fd = g_pipe_fd;
        }
        timeout.tv_sec = 0;
        timeout.tv_usec = 100000;  // 100ms timeout
        int ready = select(max_fd + 1, &readfds, NULL, NULL, &timeout);
        if (ready < 0) {
            perror("select");
            break;
        }
        // Check for Keyboard Input
        if (ready > 0 && FD_ISSET(STDIN_FILENO, &readfds)) {
            if (fgets(input, sizeof(input), stdin) == NULL) break; 
            processCommand(input);
        }
        // Check for Pipe (BLE) Input
        if (g_pipe_fd != -1 && ready > 0 && FD_ISSET(g_pipe_fd, &readfds)) {
            if (fgets(input, sizeof(input), g_pipe_stream) == NULL) {
                // EOF on pipe (python script disconnected)
                closePipe(); 
            } else {
                processCommand(input);
            }
        }
        // --- No input this cycle ---
        if (ready == 0) { 
            // Check if we need to try (re)opening the pipe
            if (g_pipe_fd == -1) {
                pipe_reconnect_timer++;
                if (pipe_reconnect_timer >= 10) { // Try to connect every 1 sec
                    pipe_reconnect_timer = 0;
                    openPipe();
                }
            }
            // Print current RPM
            pthread_mutex_lock(&g_rpm_mutex);
            double rpm = g_current_rpm;
            pthread_mutex_unlock(&g_rpm_mutex);
            if (g_pipe_fd != -1) printf("\r[BLE Ready] RPM: %7.2f | > ", rpm);
            else printf("\r[BLE Wait]  RPM: %7.2f | > ", rpm);
            fflush(stdout);
        }
    }
    printf("\nExiting main loop.\n");
    return 0;

class GameScene: SKScene, SKPhysicsContactDelegate {

    var gameOver = false
    // Category bitmasks for collision detection
    let propellerCategory: UInt32 = 0x1 << 0
    let obstacleCategory: UInt32  = 0x1 << 1
    let scoreCategory: UInt32     = 0x1 << 2
    override func didMove(to view: SKView) {
        physicsWorld.contactDelegate = self
        setupPropeller()
        setupGround()
        setupScoreLabel()
    }
    func setupPropeller() {
        propeller = SKSpriteNode(imageNamed: "propeller")
        propeller.position = CGPoint(x: frame.midX / 2, y: frame.midY)
        propeller.physicsBody = SKPhysicsBody(circleOfRadius: propeller.size.width / 2)
        propeller.physicsBody?.affectedByGravity = false
        propeller.physicsBody?.allowsRotation = true
        propeller.physicsBody?.categoryBitMask = propellerCategory
        propeller.physicsBody?.contactTestBitMask = obstacleCategory | scoreCategory
        propeller.physicsBody?.collisionBitMask = obstacleCategory
        addChild(propeller)
    }
    func setupGround() {
        let ground = SKNode()
        ground.position = CGPoint(x: 0, y: 0)
        ground.physicsBody = SKPhysicsBody(edgeLoopFrom: frame)
        ground.physicsBody?.categoryBitMask = obstacleCategory
        addChild(ground)
    }
    func setupScoreLabel() {
        scoreLabel = SKLabelNode(fontNamed: "AvenirNext-Bold")
        scoreLabel.fontSize = 40
        scoreLabel.text = "0"

        scoreLabel.zPosition = 5
        addChild(scoreLabel)

        let obstacleWidth = CGFloat(60)
        let randomHeight = CGFloat.random(in: 100...400)
        let bottomObstacle = SKSpriteNode(color: .gray, size: CGSize(width: obstacleWidth, height: randomHeight))
        bottomObstacle.position = CGPoint(x: frame.width + obstacleWidth, y: bottomObstacle.size.height / 2)
        bottomObstacle.physicsBody = SKPhysicsBody(rectangleOf: bottomObstacle.size)
        bottomObstacle.physicsBody?.isDynamic = false
        bottomObstacle.physicsBody?.categoryBitMask = obstacleCategory
        let topObstacleHeight = frame.height - randomHeight - gapHeight
        let topObstacle = SKSpriteNode(color: .gray, size: CGSize(width: obstacleWidth, height: topObstacleHeight))
        topObstacle.position = CGPoint(x: frame.width + obstacleWidth, y: frame.height - topObstacle.size.height / 2)
        topObstacle.physicsBody = SKPhysicsBody(rectangleOf: topObstacle.size)
        topObstacle.physicsBody?.isDynamic = false

        let scoreNode = SKNode()

        let move = SKAction.moveBy(x: -frame.width - 200, y: 0, duration: 4)

TECHNICAL HIGHLIGHTS:
----------------------------------------------------------------------------------------------------
• Asynchronous D-Bus calls solved C BLE timeout issues
• GPIO 5 vs GPIO 18 discovery - hardware PWM pins have interrupt limitations
• Named pipes for IPC - simple, debuggable, reliable communication
• Hardware PWM via pigpio - smooth, silent motor operation
• 500ms stabilization delay - prevents PID oscillations in automatic mode
• Motor wiggle on connect - fun UX feedback (like a dog wagging its tail!)
• Direction toggle in game - motor reverses on each pipe score

LESSONS LEARNED:
----------------------------------------------------------------------------------------------------
1. Start simple - named pipes beat sockets for basic IPC
2. Use high-level APIs when possible - avoid raw HCI if BlueZ works
3. Safety is critical - multiple fail-safes saved us from runaway motor
4. Not all GPIO pins are equal - check hardware functions before use
5. Test on real hardware early - simulators can't catch everything
6. Asynchronous > Synchronous for reliability in BLE communication
7. Document as you go - this log proves every decision
8. Git commit everything - easy rollback when experiments fail
9. PID control needs careful tuning - stabilization delays matter
10. AI generates code, humans steer direction - collaboration works!
